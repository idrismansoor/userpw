const SHEET_NAME = "Sheet1"; function doGet() { const sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName(SHEET_NAME); const data = sheet.getDataRange().getValues(); // Header বাদে এবং শুধু সেই রো গুলো নিবে যেখানে কোম্পানির নাম খালি নয় const rows = data.slice(1) .filter(row => row[1] !== "") .map(row => ({ company: row[1] || "", user: row[2] || "", password: row[3] || "", note: row[4] || "" })); return ContentService.createTextOutput(JSON.stringify({status:"ok", rows})) .setMimeType(ContentService.MimeType.JSON); } function doPost(e) { const sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName(SHEET_NAME); const body = JSON.parse(e.postData.contents); if (body.action === "save") { // শুধু ডাটা মুছে নতুন করে বসাবে, শিরোনাম ঠিক থাকবে sheet.getRange(2, 1, Math.max(sheet.getLastRow() - 1, 1), 5).clearContent(); body.rows.forEach((r, i) => { sheet.appendRow([i + 1, r.company, r.user, r.password, r.note]); }); return ContentService.createTextOutput(JSON.stringify({status:"ok"})) .setMimeType(ContentService.MimeType.JSON); } }
